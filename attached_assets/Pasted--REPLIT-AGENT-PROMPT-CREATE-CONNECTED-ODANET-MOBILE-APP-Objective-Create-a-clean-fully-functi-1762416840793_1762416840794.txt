  baseURL: process.env.EXPO_PUBLIC_API_URL || "https://www.odanet.com.tr/api",

In /hooks/useListings.ts:

import { useQuery } from "@tanstack/react-query";
import { api } from "../lib/api";

export function useListings() {
  return useQuery({
    queryKey: ["listings"],
    queryFn: async () => {
      const { data } = await api.get("/listings");
      return data;
    },
  });
}

import { View, Text, FlatList } from "react-native";
import { useListings } from "../hooks/useListings";

export default function Home() {
  const { data, isLoading, error } = useListings();

  if (isLoading) return <Text>Loading...</Text>;
  if (error) return <Text>Error loading listings</Text>;

  return (
    <View className="p-4">
      <Text className="text-xl font-bold mb-4">Available Rooms</Text>
      <FlatList
        data={data}
        keyExtractor={(item) => item.id}
        renderItem={({ item }) => (
          <View className="bg-white rounded-xl p-4 mb-3 shadow">
            <Text className="text-lg font-semibold">{item.title}</Text>
            <Text className="text-gray-500">{item.location}</Text>
          </View>
        )}
      />
    </View>
  );
}
